Step 4: Heuristic Margin Calculation on Sync
File: server/scrip-master-sync.ts (or smc-kotak-neo-v3.ts depending on where runScripMasterSync lives)
Create a new function calculatePlanMargins(storage: IStorage) and call it at the very end of runScripMasterSync, just before returning { success: true, synced }.

TypeScript
async function calculatePlanMargins(storage: IStorage) {
  try {
    const plans = await storage.getAllStrategyPlans();
    
    // Default heuristic values (these can be moved to settings later)
    const BASE_SELL_MARGIN = 100000;
    const HEDGE_DISCOUNT = 60000;
    const SAFE_BUY_MARGIN = 4000; // Rough estimate for buying 1 lot
    
    for (const plan of plans) {
       if (plan.deploymentStatus !== 'active' && plan.deploymentStatus !== 'deployed') continue;
       
       let totalMargin = 0;
       let sellLots = 0;
       let buyLots = 0;
       
       try {
           const params = typeof plan.tradeParams === 'string' ? JSON.parse(plan.tradeParams) : plan.tradeParams;
           if (!params) continue;
           
           const allLegs = [
               ...(params.uptrendLegs || []),
               ...(params.downtrendLegs || []),
               ...(params.neutralLegs || [])
           ];
           
           const lotMultiplier = plan.lotMultiplier || 1;
           
           for (const leg of allLegs) {
               const qty = (leg.lots || 1) * lotMultiplier;
               if (leg.action === 'BUY') {
                   totalMargin += (SAFE_BUY_MARGIN * qty); 
                   buyLots += qty;
               } else if (leg.action === 'SELL') {
                   totalMargin += (BASE_SELL_MARGIN * qty);
                   sellLots += qty;
               }
           }
           
           // Apply Heuristic Hedge Discount for paired lots
           const hedgedPairs = Math.min(sellLots, buyLots);
           totalMargin -= (hedgedPairs * HEDGE_DISCOUNT);
           totalMargin = Math.max(totalMargin, 0);
           
           await storage.updateStrategyPlan(plan.id, {
               estimatedMargin: totalMargin.toString(),
               marginCalculatedAt: new Date().toISOString()
           });
           
       } catch (e) {
           console.error(`[SMC] Error calculating margin for plan ${plan.id}:`, e);
       }
    }
  } catch (err) {
    console.error("[SMC] Failed to execute calculatePlanMargins:", err);
  }
}

  const activePlans = plans.filter(p => p.autoResume !== false);
  activePlans.sort((a, b) => (a.rank || 999) - (b.rank || 999));
  
  // Determine available capital (can pull from settings, defaulting to 10L)
  const maxCapSetting = await storage.getSetting("maximum_allocated_capital");
  let availableCapital = maxCapSetting?.value ? parseFloat(maxCapSetting.value) : 1000000;
  
  const fundedPlans = [];
  for (const plan of activePlans) {
      const requiredMargin = plan.estimatedMargin ? parseFloat(plan.estimatedMargin as string) : 0;
      if (availableCapital >= requiredMargin) {
          fundedPlans.push(plan);
          availableCapital -= requiredMargin;
      } else {
          console.warn(`[TE] Skipping Plan ${plan.name} (Rank ${plan.rank}) due to insufficient margin. Req: ${requiredMargin}, Avail: ${availableCapital}`);
          logPFL(plan, "system", webhookData, "skipped", `Insufficient margin (Req: ₹${requiredMargin}, Avail: ₹${availableCapital})`);
      }
  }
