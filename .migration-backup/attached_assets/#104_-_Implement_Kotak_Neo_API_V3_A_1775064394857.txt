Task 1 — Add V3 Body Formatter and Relay Wrapper
File: server/el-kotak-neo-v3.ts
Locate the ExecutionLayer class. Add these two private helpers to handle V3 formatting and the Bangalore relay routing:

TypeScript
  /**
   * Formats the body for Kotak V3 API. V3 requires POST payloads to be URL-encoded 
   * under the key "jData".
   */
  private formatV3Body(body: any): { body: string; contentType?: string } {
    if (!body) return { body: "" };
    
    // Convert the JSON object to a string, then URL-encode it, prefixing with jData=
    const jDataString = JSON.stringify(body);
    const encodedBody = `jData=${encodeURIComponent(jDataString)}`;
    
    return {
      body: encodedBody,
      contentType: "application/x-www-form-urlencoded"
    };
  }

  /**
   * Routes Kotak API requests through the Bangalore Static IP Relay.
   * Required for SEBI API IP Whitelisting and Session Binding.
   */
  private async executeRelayFetch(targetUrl: string, options: any): Promise<Response> {
    const RELAY_URL = "http://168.144.0.191:8080";
    const RELAY_SECRET = "mentorsworld_algo_secure_2026";

    // Strip "Bearer " from Authorization header if present (Kotak V3 rejects it)
    if (options?.headers?.["Authorization"]) {
      options.headers["Authorization"] = options.headers["Authorization"].replace(/bearer /i, "").trim();
    }

    const relayOptions = {
      ...options,
      headers: {
        ...options?.headers,
        'x-target-url': targetUrl,
        'x-relay-secret': RELAY_SECRET
      }
    };

    return fetch(RELAY_URL, relayOptions);
  }
