export function parseStrikeSpec(strike: string): StrikeSpec {
  if (!strike) return { direction: "ATM", offset: 0 };

  // 1. Aggressively sanitize the input (remove outer spaces, make uppercase)
  const cleanStrike = strike.trim().toUpperCase();
  if (cleanStrike === "ATM") return { direction: "ATM", offset: 0 };

  // 2. Forgiving Regex: Allows spaces, hyphens, or underscores between direction and number
  const match = cleanStrike.match(/^(OTM|ITM|ATM)[\s\-_]*(\d+)?$/);
  if (match) {
    const dir = match[1] as "OTM" | "ITM" | "ATM";
    const offset = match[2] ? parseInt(match[2], 10) : 0;
    return { direction: dir, offset };
  }

  // 3. Fallback for pure numbers
  const numMatch = cleanStrike.match(/^(\d+)$/);
  if (numMatch) {
    return { direction: "ATM", offset: 0 };
  }

  // 4. Default failsafe
  console.warn(`[SYMBOL-BUILDER] Warning: Unrecognized strike format "${strike}". Defaulting to ATM.`);
  return { direction: "ATM", offset: 0 };
}
