I completely understand the architecture you are aiming for now. You are building for massive multi-tenant scale (1000+ users).

If you tie the Margin Calculator directly to the SMC memory cache, you risk bottlenecking the server during the morning sync, especially if 1000+ strategies are trying to read a single memory map simultaneously while the CSV is still parsing.

You want cm-kotak-neo-v3.ts to be completely autonomous:

Independent Token Resolution: It should figure out the ATM strike and grab the exact token on its own (likely by querying your database's instrumentConfigs or using the broker API directly).

Independent Margin Check: It pings the Kotak REST API.

Independent Capital Check: It pings Kotak's getLimits REST API to compare required margin vs. actual funds available.

Because we are decoupling this, the CM file will act as a standalone Risk Management Daemon.

export async function processCapitalRequirements(storage: IStorage) {
  console.log(`${LOG_PREFIX} Starting autonomous capital management sweep for all users...`);
  
  try {
    const plans = await storage.getStrategyPlans();
    const brokerConfigsList = await storage.getBrokerConfigs();
    const brokerMap = new Map(brokerConfigsList.map(bc => [bc.id, bc]));

    // Base fallback heuristics in case API fails
    const baseSellMargin = 100000;
    const hedgeDiscount = 60000;
    const safeBuyMargin = 4000;

    for (const plan of plans) {
      if (plan.deploymentStatus !== 'active' && plan.deploymentStatus !== 'deployed') continue;
      
      const brokerConfig = brokerMap.get(plan.brokerConfigId);
      if (!brokerConfig || !brokerConfig.isConnected) continue;

      try {
        // 1. Fetch Live Available Funds
        const limitsRes = await EL.getLimits(brokerConfig);
        const availableFunds = limitsRes.success && limitsRes.data ? parseFloat(limitsRes.data.avlCash || limitsRes.data.avlMrgn || "0") : 0;

        // 2. Calculate Required Margin
        const params = typeof plan.tradeParams === 'string' ? JSON.parse(plan.tradeParams) : plan.tradeParams;
        if (!params) continue;

        const allLegs = [
          ...(params.uptrendLegs || []),
          ...(params.downtrendLegs || []),
          ...(params.neutralLegs || [])
        ];

        const lotMultiplier = plan.lotMultiplier || 1;
        const instrConfig = getInstrumentConfig(plan.ticker, plan.exchange || "NFO");
        const lotSize = instrConfig?.lotSize || 25;
        const strikeInterval = instrConfig?.strikeInterval || 50;

        const quoteRes = await EL.getQuote(brokerConfig, EL.mapExchange(plan.exchange || "NFO"), plan.ticker);
        const ltp = quoteRes.success && quoteRes.ltp ? quoteRes.ltp : null;

        let totalMarginAPI = 0;
        let apiSuccess = true;
        let sellLots = 0;
        let buyLots = 0;
        let totalMarginInternal = 0;

        const atmStrike = ltp ? Math.round(ltp / strikeInterval) * strikeInterval : 0;

        for (const leg of allLegs) {
          const qty = (leg.lots || 1) * lotMultiplier;
          const totalLegQty = qty * lotSize;
            
          if (leg.action === 'BUY') {
              totalMarginInternal += (safeBuyMargin * qty); 
              buyLots += qty;
          } else if (leg.action === 'SELL') {
              totalMarginInternal += (baseSellMargin * qty);
              sellLots += qty;
          }

          if (atmStrike > 0 && apiSuccess) {
              const offset = leg.strikeOffset || 0;
              const targetStrike = leg.optionType === "CE" ? atmStrike + (offset * strikeInterval) : atmStrike - (offset * strikeInterval);

              // AUTONOMOUS TOKEN RESOLUTION
              const token = await resolveTokenAutonomous(storage, plan.ticker, targetStrike, leg.optionType);

              if (token) {
                  const marginRes = await EL.checkMargin(brokerConfig, {
                    exSeg: EL.mapExchange(plan.exchange || "NFO"),
                    prc: "0",
                    prcTp: "MKT",
                    prod: params.blockConfig?.productMode || "MIS",
                    qty: String(totalLegQty),
                    tok: token,
                    trnsTp: leg.action === "BUY" ? "B" : "S",
                    brkName: "KOTAK",
                    brnchId: "ONLINE"
                  });

                  if (marginRes.success && marginRes.data && marginRes.data.ordMrgn) {
                    totalMarginAPI += parseFloat(marginRes.data.ordMrgn);
                  } else {
                    apiSuccess = false; 
                  }
              } else {
                  apiSuccess = false;
              }
          } else {
              apiSuccess = false;
          }
        }

        // 3. Final Margin Determination
        let finalMargin = 0;
        if (apiSuccess && totalMarginAPI > 0) {
            finalMargin = totalMarginAPI;
        } else {
            const hedgedPairs = Math.min(sellLots, buyLots);
            totalMarginInternal -= (hedgedPairs * hedgeDiscount);
            finalMargin = Math.max(totalMarginInternal, 0);
        }

        // 4. Update Database
        if (finalMargin > 0) {
          await storage.updateStrategyPlan(plan.id, {
            estimatedMargin: finalMargin.toFixed(2),
            marginCalculatedAt: new Date().toISOString()
          });
          
          // Cross-Check Log (Simulating the 1000+ user requirement)
          if (availableFunds < finalMargin) {
              console.warn(`${LOG_PREFIX} CAUTION: User ${brokerConfig.id} has insufficient funds for Plan ${plan.name}. Required: ₹${finalMargin}, Available: ₹${availableFunds}`);
          }
        }
      } catch (err) {
        console.error(`${LOG_PREFIX} Error processing plan ${plan.id}:`, err);
      }
    }
  } catch (globalErr) {
    console.error(`${LOG_PREFIX} Critical failure in CM Engine:`, globalErr);
  }
}
