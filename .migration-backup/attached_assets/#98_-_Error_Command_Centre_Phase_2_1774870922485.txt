Task 3 — The Central Switchboard
File: server/te-kotak-neo-v3.ts
Add this function at the top level of the file. Note: ensure storage.getActiveErrorRoutes() is implemented in storage.ts.

TypeScript
export async function handleBrokerError(errorStr: string, context: { trade?: StrategyTrade, plan?: StrategyPlan }) {
  const activeRoutes = await storage.getActiveErrorRoutes(); 
  const errStrLower = String(errorStr || "").toLowerCase();
  
  const matchedRoute = activeRoutes.find((route: any) => 
    errStrLower.includes(route.errorPattern.toLowerCase())
  );

  if (!matchedRoute) return false; // Not mapped, fall back to existing retry loops

  switch (matchedRoute.actionType) {
    case "system_halt":
      const haltMsg = `CRITICAL: System halted due to Auth error: ${errorStr}`;
      console.error(haltMsg);
      await storage.upsertSetting("trading_halted", "true");
      if (context.plan) await addProcessFlowLog(context.plan.id, "N/A", "system_halt", haltMsg);
      return "system_halt";

    case "cancel_plan":
      if (context.plan) {
        const cancelMsg = `Plan aborted due to terminal entry error: ${errorStr}`;
        console.warn(cancelMsg);
        await storage.updateStrategyPlan(context.plan.id, { deploymentStatus: "paused", status: "failed" });
        await addProcessFlowLog(context.plan.id, "N/A", "plan_aborted", cancelMsg);
      }
      return "cancel_plan";

    case "terminal_close":
      if (context.trade && context.plan) {
        const terminalMsg = `[TE] Terminal rejection mapped by rule #${matchedRoute.id}: "${errorStr}". Marking as closed.`;
        console.warn(terminalMsg);
        await addProcessFlowLog(context.plan.id, context.trade.entryLegId || "N/A", "force_closed", terminalMsg);
        await storage.updateStrategyTrade(context.trade.id, {
          status: "closed", pnl: 0, exitedAt: new Date(), updatedAt: new Date(), 
          exitPrice: context.trade.ltp || context.trade.price || 0, exitAction: context.plan.exitAction || "SELL"
        });
        tradingCache.invalidateOpenTrades(context.plan.id);
      }
      return "terminal_close";

    default:
      return false;
  }
}
