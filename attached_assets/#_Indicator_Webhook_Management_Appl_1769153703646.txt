## Database Schema

### Tables Created

#### 1. `webhooks` Table
Stores webhook configurations and metadata.

```sql
CREATE TABLE webhooks (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  description TEXT,
  url TEXT NOT NULL,
  secret_key TEXT NOT NULL,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- RLS Policies
ALTER TABLE webhooks ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can view their own webhooks" 
  ON webhooks FOR SELECT 
  USING (auth.uid() = user_id);

CREATE POLICY "Users can insert their own webhooks" 
  ON webhooks FOR INSERT 
  WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Users can update their own webhooks" 
  ON webhooks FOR UPDATE 
  USING (auth.uid() = user_id);

CREATE POLICY "Users can delete their own webhooks" 
  ON webhooks FOR DELETE 
  USING (auth.uid() = user_id);
```

#### 2. `webhook_logs` Table
Stores all webhook execution logs and payloads.

```sql
CREATE TABLE webhook_logs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  webhook_id UUID NOT NULL REFERENCES webhooks(id) ON DELETE CASCADE,
  status TEXT NOT NULL,
  payload JSONB NOT NULL,
  response_data JSONB,
  error_message TEXT,
  ip_address TEXT,
  user_agent TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Index for faster queries
CREATE INDEX idx_webhook_logs_webhook_id ON webhook_logs(webhook_id);
CREATE INDEX idx_webhook_logs_created_at ON webhook_logs(created_at DESC);
CREATE INDEX idx_webhook_logs_status ON webhook_logs(status);

-- RLS Policies
ALTER TABLE webhook_logs ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can view logs for their webhooks" 
  ON webhook_logs FOR SELECT 
  USING (
    EXISTS (
      SELECT 1 FROM webhooks 
      WHERE webhooks.id = webhook_logs.webhook_id 
      AND webhooks.user_id = auth.uid()
    )
  );

CREATE POLICY "Allow insert webhook logs" 
  ON webhook_logs FOR INSERT 
  WITH CHECK (true);
```

## API Routes

### 1. Webhook Receiver API - `/api/webhook/[id].ts`

This is the main webhook endpoint that receives POST requests from TradingView or other platforms.

```typescript
import type { NextApiRequest, NextApiResponse } from "next";
import { supabase } from "@/integrations/supabase/client";
import type { Database } from "@/integrations/supabase/types";

type WebhookLog = Database["public"]["Tables"]["webhook_logs"]["Insert"];

export default async function handler(
  req: NextApiRequest,
  res: NextApiResponse
) {
  // Only allow POST requests
  if (req.method !== "POST") {
    return res.status(405).json({ error: "Method not allowed" });
  }

  const webhookId = req.query.id as string;

  if (!webhookId) {
    return res.status(400).json({ error: "Webhook ID is required" });
  }

  try {
    console.log(`[Webhook ${webhookId}] Received request`);
    console.log(`[Webhook ${webhookId}] Headers:`, req.headers);
    console.log(`[Webhook ${webhookId}] Body:`, req.body);

    // Get webhook configuration
    const { data: webhook, error: webhookError } = await supabase
      .from("webhooks")
      .select("*")
      .eq("id", webhookId)
      .single();

    if (webhookError || !webhook) {
      console.error(`[Webhook ${webhookId}] Not found:`, webhookError);
      await logWebhookExecution(webhookId, "failed", req.body, null, "Webhook not found", req);
      return res.status(404).json({ error: "Webhook not found" });
    }

    // Check if webhook is active
    if (!webhook.is_active) {
      console.log(`[Webhook ${webhookId}] Inactive`);
      await logWebhookExecution(webhookId, "failed", req.body, null, "Webhook is inactive", req);
      return res.status(403).json({ error: "Webhook is inactive" });
    }

    // Verify secret key if provided in headers
    const providedSecret = req.headers["x-webhook-secret"] as string;
    if (providedSecret && providedSecret !== webhook.secret_key) {
      console.error(`[Webhook ${webhookId}] Invalid secret key`);
      await logWebhookExecution(webhookId, "failed", req.body, null, "Invalid secret key", req);
      return res.status(401).json({ error: "Invalid secret key" });
    }

    // Process webhook payload
    const payload = req.body;

    // Here you can add custom logic to process the webhook
    // For example: place orders, trigger strategies, send notifications, etc.
    console.log(`[Webhook ${webhookId}] Processing payload:`, payload);

    // Log successful execution
    await logWebhookExecution(webhookId, "success", payload, { processed: true }, null, req);

    return res.status(200).json({
      success: true,
      message: "Webhook received and processed successfully",
      webhookId,
      timestamp: new Date().toISOString(),
    });
  } catch (error) {
    console.error(`[Webhook ${webhookId}] Error:`, error);
    await logWebhookExecution(
      webhookId,
      "failed",
      req.body,
      null,
      error instanceof Error ? error.message : "Unknown error",
      req
    );
    return res.status(500).json({
      error: "Internal server error",
      details: error instanceof Error ? error.message : "Unknown error",
    });
  }
}

// Helper function to log webhook execution
async function logWebhookExecution(
  webhookId: string,
  status: string,
  payload: any,
  responseData: any,
  errorMessage: string | null,
  req: NextApiRequest
) {
  try {
    const logEntry: WebhookLog = {
      webhook_id: webhookId,
      status,
      payload,
      response_data: responseData,
      error_message: errorMessage,
      ip_address: (req.headers["x-forwarded-for"] as string) || req.socket.remoteAddress || null,
      user_agent: req.headers["user-agent"] || null,
    };

    const { error } = await supabase.from("webhook_logs").insert(logEntry);

    if (error) {
      console.error(`[Webhook ${webhookId}] Failed to log execution:`, error);
    } else {
      console.log(`[Webhook ${webhookId}] Execution logged successfully`);
    }
  } catch (error) {
    console.error(`[Webhook ${webhookId}] Error logging execution:`, error);
  }
}
```

## Services

### Webhook Log Service - `src/services/webhookLogService.ts`

Service layer for managing webhook logs with type-safe database operations.

```typescript
import { supabase } from "@/integrations/supabase/client";
import type { Database } from "@/integrations/supabase/types";

type WebhookLog = Database["public"]["Tables"]["webhook_logs"]["Row"];
type WebhookLogInsert = Database["public"]["Tables"]["webhook_logs"]["Insert"];

export interface WebhookLogWithWebhook extends WebhookLog {
  webhooks: {
    id: string;
    name: string;
    url: string;
  } | null;
}

export interface WebhookLogFilters {
  webhookId?: string;
  status?: string;
  startDate?: string;
  endDate?: string;
  search?: string;
}

export interface WebhookLogStats {
  total: number;
  success: number;
  failed: number;
  successRate: number;
}

/**
 * Fetch webhook logs with optional filters
 */
export async function fetchWebhookLogs(
  userId: string,
  filters?: WebhookLogFilters,
  limit: number = 50,
  offset: number = 0
): Promise<{ logs: WebhookLogWithWebhook[]; count: number; error: Error | null }> {
  try {
    let query = supabase
      .from("webhook_logs")
      .select(
        `
        *,
        webhooks!inner (
          id,
          name,
          url,
          user_id
        )
      `,
        { count: "exact" }
      )
      .eq("webhooks.user_id", userId)
      .order("created_at", { ascending: false })
      .range(offset, offset + limit - 1);

    // Apply filters
    if (filters?.webhookId) {
      query = query.eq("webhook_id", filters.webhookId);
    }

    if (filters?.status) {
      query = query.eq("status", filters.status);
    }

    if (filters?.startDate) {
      query = query.gte("created_at", filters.startDate);
    }

    if (filters?.endDate) {
      query = query.lte("created_at", filters.endDate);
    }

    if (filters?.search) {
      query = query.or(
        `payload::text.ilike.%${filters.search}%,error_message.ilike.%${filters.search}%`
      );
    }

    const { data, error, count } = await query;

    if (error) {
      console.error("[WebhookLogService] Error fetching logs:", error);
      return { logs: [], count: 0, error };
    }

    return {
      logs: (data as WebhookLogWithWebhook[]) || [],
      count: count || 0,
      error: null,
    };
  } catch (error) {
    console.error("[WebhookLogService] Unexpected error:", error);
    return {
      logs: [],
      count: 0,
      error: error instanceof Error ? error : new Error("Unknown error"),
    };
  }
}

/**
 * Get webhook log statistics
 */
export async function fetchWebhookLogStats(
  userId: string,
  webhookId?: string,
  startDate?: string,
  endDate?: string
): Promise<{ stats: WebhookLogStats | null; error: Error | null }> {
  try {
    let query = supabase
      .from("webhook_logs")
      .select(
        `
        status,
        webhooks!inner (
          user_id
        )
      `
      )
      .eq("webhooks.user_id", userId);

    if (webhookId) {
      query = query.eq("webhook_id", webhookId);
    }

    if (startDate) {
      query = query.gte("created_at", startDate);
    }

    if (endDate) {
      query = query.lte("created_at", endDate);
    }

    const { data, error } = await query;

    if (error) {
      console.error("[WebhookLogService] Error fetching stats:", error);
      return { stats: null, error };
    }

    const total = data?.length || 0;
    const success = data?.filter((log) => log.status === "success").length || 0;
    const failed = data?.filter((log) => log.status === "failed").length || 0;
    const successRate = total > 0 ? (success / total) * 100 : 0;

    return {
      stats: {
        total,
        success,
        failed,
        successRate,
      },
      error: null,
    };
  } catch (error) {
    console.error("[WebhookLogService] Unexpected error:", error);
    return {
      stats: null,
      error: error instanceof Error ? error : new Error("Unknown error"),
    };
  }
}

/**
 * Delete webhook logs older than specified days
 */
export async function deleteOldWebhookLogs(
  userId: string,
  daysToKeep: number = 30
): Promise<{ deletedCount: number; error: Error | null }> {
  try {
    const cutoffDate = new Date();
    cutoffDate.setDate(cutoffDate.getDate() - daysToKeep);

    // First get the webhook IDs for this user
    const { data: webhooks, error: webhooksError } = await supabase
      .from("webhooks")
      .select("id")
      .eq("user_id", userId);

    if (webhooksError) {
      console.error("[WebhookLogService] Error fetching webhooks:", webhooksError);
      return { deletedCount: 0, error: webhooksError };
    }

    const webhookIds = webhooks?.map((w) => w.id) || [];

    if (webhookIds.length === 0) {
      return { deletedCount: 0, error: null };
    }

    // Delete old logs for these webhooks
    const { data, error } = await supabase
      .from("webhook_logs")
      .delete()
      .in("webhook_id", webhookIds)
      .lt("created_at", cutoffDate.toISOString())
      .select();

    if (error) {
      console.error("[WebhookLogService] Error deleting old logs:", error);
      return { deletedCount: 0, error };
    }

    return { deletedCount: data?.length || 0, error: null };
  } catch (error) {
    console.error("[WebhookLogService] Unexpected error:", error);
    return {
      deletedCount: 0,
      error: error instanceof Error ? error : new Error("Unknown error"),
    };
  }
}

/**
 * Create a new webhook log entry
 */
export async function createWebhookLog(
  logData: WebhookLogInsert
): Promise<{ log: WebhookLog | null; error: Error | null }> {
  try {
    const { data, error } = await supabase
      .from("webhook_logs")
      .insert(logData)
      .select()
      .single();

    if (error) {
      console.error("[WebhookLogService] Error creating log:", error);
      return { log: null, error };
    }

    return { log: data, error: null };
  } catch (error) {
    console.error("[WebhookLogService] Unexpected error:", error);
    return {
      log: null,
      error: error instanceof Error ? error : new Error("Unknown error"),
    };
  }
}
```

### 2. Webhook Logs Page - `src/pages/webhook-logs/[id].tsx`

Detailed logs page for a specific webhook (225 lines - showing key sections).

```typescript
import { useState, useEffect } from "react";
import { useRouter } from "next/router";
import { supabase } from "@/integrations/supabase/client";
import { fetchWebhookLogs, fetchWebhookLogStats } from "@/services/webhookLogService";
import type { WebhookLogWithWebhook, WebhookLogStats } from "@/services/webhookLogService";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { useToast } from "@/hooks/use-toast";
import { ArrowLeft, RefreshCw, CheckCircle2, XCircle, Activity } from "lucide-react";

export default function WebhookLogsPage() {
  const router = useRouter();
  const { id } = router.query;
  const { toast } = useToast();

  const [user, setUser] = useState<any>(null);
  const [logs, setLogs] = useState<WebhookLogWithWebhook[]>([]);
  const [stats, setStats] = useState<WebhookLogStats | null>(null);
  const [loading, setLoading] = useState(true);
  const [webhookName, setWebhookName] = useState("");

  useEffect(() => {
    loadUser();
  }, []);

  useEffect(() => {
    if (user && id) {
      loadLogsAndStats();
    }
  }, [user, id]);

  const loadUser = async () => {
    try {
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) {
        router.push("/login");
        return;
      }
      setUser(user);
    } catch (error) {
      console.error("Error loading user:", error);
    }
  };

  const loadLogsAndStats = async () => {
    if (!user || !id) return;

    setLoading(true);
    try {
      // Load logs
      const { logs: fetchedLogs, error: logsError } = await fetchWebhookLogs(
        user.id,
        { webhookId: id as string },
        100,
        0
      );

      if (logsError) throw logsError;

      setLogs(fetchedLogs);
      if (fetchedLogs.length > 0 && fetchedLogs[0].webhooks) {
        setWebhookName(fetchedLogs[0].webhooks.name);
      }

      // Load stats
      const { stats: fetchedStats, error: statsError } = await fetchWebhookLogStats(
        user.id,
        id as string
      );

      if (statsError) throw statsError;

      setStats(fetchedStats);
    } catch (error) {
      console.error("Error loading logs:", error);
      toast({
        title: "Error",
        description: "Failed to load webhook logs",
        variant: "destructive",
      });
    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <RefreshCw className="h-8 w-8 animate-spin text-blue-600" />
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-50 to-blue-50 p-6">
      <div className="max-w-7xl mx-auto space-y-6">
        {/* Header */}
        <div className="flex items-center gap-4">
          <Button variant="outline" size="sm" onClick={() => router.push("/webhooks")}>
            <ArrowLeft className="h-4 w-4" />
          </Button>
          <div className="flex-1">
            <h1 className="text-3xl font-bold text-slate-900">Webhook Logs</h1>
            <p className="text-slate-600 mt-1">{webhookName || "Loading..."}</p>
          </div>
          <Button variant="outline" size="sm" onClick={loadLogsAndStats}>
            <RefreshCw className="h-4 w-4" />
          </Button>
        </div>

        {/* Stats Cards */}
        {stats && (
          <div className="grid gap-4 md:grid-cols-4">
            <Card>
              <CardHeader className="pb-2">
                <CardTitle className="text-sm font-medium text-slate-600">
                  Total Requests
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="text-2xl font-bold">{stats.total}</div>
              </CardContent>
            </Card>
            <Card>
              <CardHeader className="pb-2">
                <CardTitle className="text-sm font-medium text-slate-600">
                  Successful
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="text-2xl font-bold text-green-600">{stats.success}</div>
              </CardContent>
            </Card>
            <Card>
              <CardHeader className="pb-2">
                <CardTitle className="text-sm font-medium text-slate-600">
                  Failed
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="text-2xl font-bold text-red-600">{stats.failed}</div>
              </CardContent>
            </Card>
            <Card>
              <CardHeader className="pb-2">
                <CardTitle className="text-sm font-medium text-slate-600">
                  Success Rate
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="text-2xl font-bold">
                  {stats.successRate.toFixed(1)}%
                </div>
              </CardContent>
            </Card>
          </div>
        )}

        {/* Logs Table */}
        <Card>
          <CardHeader>
            <CardTitle>Execution Logs</CardTitle>
          </CardHeader>
          <CardContent>
            {logs.length === 0 ? (
              <div className="text-center py-12 text-slate-500">
                <Activity className="h-12 w-12 mx-auto mb-4 opacity-50" />
                <p>No logs found for this webhook</p>
              </div>
            ) : (
              <div className="space-y-4">
                {logs.map((log) => (
                  <div
                    key={log.id}
                    className="border rounded-lg p-4 hover:bg-slate-50 transition-colors"
                  >
                    <div className="flex items-start justify-between mb-2">
                      <div className="flex items-center gap-2">
                        {log.status === "success" ? (
                          <CheckCircle2 className="h-5 w-5 text-green-600" />
                        ) : (
                          <XCircle className="h-5 w-5 text-red-600" />
                        )}
                        <Badge variant={log.status === "success" ? "default" : "destructive"}>
                          {log.status}
                        </Badge>
                      </div>
                      <span className="text-sm text-slate-500">
                        {new Date(log.created_at).toLocaleString()}
                      </span>
                    </div>

                    {/* Payload */}
                    <div className="mt-2">
                      <p className="text-xs text-slate-500 mb-1">Payload:</p>
                      <pre className="text-xs bg-slate-100 p-2 rounded overflow-x-auto">
                        {JSON.stringify(log.payload, null, 2)}
                      </pre>
                    </div>

                    {/* Error Message */}
                    {log.error_message && (
                      <div className="mt-2">
                        <p className="text-xs text-slate-500 mb-1">Error:</p>
                        <p className="text-xs text-red-600 bg-red-50 p-2 rounded">
                          {log.error_message}
                        </p>
                      </div>
                    )}

                    {/* Meta Info */}
                    <div className="mt-2 flex gap-4 text-xs text-slate-500">
                      {log.ip_address && <span>IP: {log.ip_address}</span>}
                      {log.user_agent && (
                        <span className="truncate max-w-xs">
                          Agent: {log.user_agent}
                        </span>
                      )}
                    </div>
                  </div>
                ))}
              </div>
            )}
          </CardContent>
        </Card>
      </div>
    </div>
  );
}
```

### Example 1: TradingView Alert Setup

```json
{
  "symbol": "{{ticker}}",
  "price": "{{close}}",
  "time": "{{time}}",
  "interval": "{{interval}}",
  "action": "{{strategy.order.action}}",
  "contracts": "{{strategy.order.contracts}}"
}
```
- In Webhook URL field, paste your webhook URL
- Add header: `x-webhook-secret: YOUR_SECRET_KEY`
- Save alert

**3. Test the Webhook:**
- Go back to your application
- Click "Test" button on the webhook card
- Check logs to verify it's working
