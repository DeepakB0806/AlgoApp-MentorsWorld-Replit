// ─── Schema-Driven Distance-Based SPAN Block ───────────────────────────────────
async function cmDistanceSpanBlock(
  legs: any[],
  atmStrike: number,
  strikeInterval: number,
  tokenMap: Map<string, string>,
  priceMap: Map<number, { CE?: number; PE?: number }>,
  effectiveSpanRate: number,
  lotSize: number, // DYNAMICALLY PASSED FROM SCHEMA
  lotMultiplier: number, // DYNAMICALLY PASSED FROM SCHEMA
  brokerConfig: BrokerConfig,
  exchange: string,
  blockLabel: string,
): Promise<number> {
  const MLOG = "[MARGIN-CALC]";
  let blockMargin = 0;
  
  // Base Exposure Rate (~2% for Indices)
  const exposureRate = 0.02; 
  const contractValue = atmStrike * lotSize;
  const nakedSellMarginPerLot = contractValue * (effectiveSpanRate + exposureRate);

  const calls = { sells: [] as any[], buys: [] as any[] };
  const puts = { sells: [] as any[], buys: [] as any[] };

  for (const leg of legs) {
    const legType = (leg.type || "").toUpperCase();
    const action = (leg.action || "SELL").toUpperCase();
    
    let resolvedStrike = atmStrike;
    if (legType !== "FUT") {
        const spec = parseStrikeSpec(leg.strike || "ATM");
        resolvedStrike = getOTMStrike(atmStrike, spec, strikeInterval, legType as "CE" | "PE");
    }

    let token = legType === "FUT" ? tokenMap.get(`0_FUT`) : tokenMap.get(`${resolvedStrike}_${legType}`);
    
    let premium = 0;
    let priceSource = "CSV-fallback";
    if (token && brokerConfig.isConnected && brokerConfig.accessToken) {
        try {
            const quoteRes = await EL.getQuote(brokerConfig, EL.mapExchange(exchange), token);
            if (quoteRes.success && quoteRes.ltp && quoteRes.ltp > 0) {
                premium = quoteRes.ltp;
                priceSource = "LIVE-API";
                await new Promise(r => setTimeout(r, 50)); 
            }
        } catch (err) {}
    }
    if (premium === 0) {
        const priceEntry = priceMap.get(resolvedStrike);
        premium = legType === "FUT" ? resolvedStrike : ((legType === "CE" ? priceEntry?.CE : priceEntry?.PE) ?? 150);
        if (premium === 150) priceSource = "DEFAULT-150";
    }

    const processedLeg = { ...leg, resolvedStrike, premium, priceSource };

    if (legType === "CE") {
        action === "SELL" ? calls.sells.push(processedLeg) : calls.buys.push(processedLeg);
    } else if (legType === "PE") {
        action === "SELL" ? puts.sells.push(processedLeg) : puts.buys.push(processedLeg);
    } else if (legType === "FUT") {
        blockMargin += action === "SELL" ? nakedSellMarginPerLot : (premium * lotSize);
    }
  }

  const processPairs = (sells: any[], buys: any[], typeLabel: string) => {
      let margin = 0;
      sells.sort((a, b) => (a.strikeOffset || 0) - (b.strikeOffset || 0));
      buys.sort((a, b) => (a.strikeOffset || 0) - (b.strikeOffset || 0));

      let buyIdx = 0;
      for (const sell of sells) {
          // SCHEMA FETCH: read lots directly from the leg schema
          let sellLots = (sell.lots || 1) * lotMultiplier; 
          
          while (sellLots > 0) {
              if (buyIdx < buys.length) {
                  const buy = buys[buyIdx];
                  // SCHEMA FETCH: read lots directly from the leg schema
                  let buyLots = (buy.lots || 1) * lotMultiplier;
                  
                  const strikeDiff = Math.abs((sell.resolvedStrike || 0) - (buy.resolvedStrike || 0));
                  const spreadMarginPerLot = (strikeDiff * lotSize) + (contractValue * exposureRate);
                  
                  const pairs = Math.min(sellLots, buyLots);
                  margin += (spreadMarginPerLot * pairs);
                  console.log(`${MLOG}   ${blockLabel} SPAN: HEDGED ${typeLabel} S@${sell.resolvedStrike}/B@${buy.resolvedStrike} req=₹${(spreadMarginPerLot * pairs).toFixed(2)}`);

                  sellLots -= pairs;
                  buyLots -= pairs;
                  if (buyLots <= 0) buyIdx++;
              } else {
                  margin += (nakedSellMarginPerLot * sellLots);
                  console.log(`${MLOG}   ${blockLabel} SPAN: NAKED SELL ${typeLabel} @${sell.resolvedStrike} req=₹${(nakedSellMarginPerLot * sellLots).toFixed(2)}`);
                  sellLots = 0;
              }
          }
      }

      while (buyIdx < buys.length) {
           const buy = buys[buyIdx];
           const leftoverLots = (buy.lots || 1) * lotMultiplier;
           const buyReq = buy.premium * lotSize * leftoverLots;
           margin += buyReq;
           console.log(`${MLOG}   ${blockLabel} SPAN: NAKED BUY ${typeLabel} @${buy.resolvedStrike} prem=₹${buy.premium.toFixed(2)} req=₹${buyReq.toFixed(2)}`);
           buyIdx++;
      }
      return margin;
  };

  blockMargin += processPairs(calls.sells, calls.buys, "CE");
  blockMargin += processPairs(puts.sells, puts.buys, "PE");

  return blockMargin;
}
