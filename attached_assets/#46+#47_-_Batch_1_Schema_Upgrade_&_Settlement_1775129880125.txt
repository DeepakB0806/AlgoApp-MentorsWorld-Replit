Task 2 — Create Settlement Engine (#47)
File: server/se-kotak.ts (Create new)
Build a module that safely processes the math for closed trades.

TypeScript
import { db } from "./db";
import { strategyTrades, strategies } from "@shared/schema";
import { eq, and, inArray } from "drizzle-orm";
import { log } from "./process-flow-log";

export class SettlementEngine {
  public async processUnsettledTrades() {
    try {
      // 1. Find all closed trades where P&L hasn't been added to strategy totals
      const unsettled = await db.select().from(strategyTrades)
        .where(and(eq(strategyTrades.status, "CLOSED"), eq(strategyTrades.pnlCalculated, false)));

      if (!unsettled.length) return;

      for (const trade of unsettled) {
        // 2. Fetch the parent strategy
        const strategyList = await db.select().from(strategies).where(eq(strategies.id, trade.strategyId));
        if (!strategyList.length) continue;
        const strategy = strategyList[0];

        // 3. Math: Add trade PnL to Strategy Daily and Total
        const tradePnl = trade.realizedPnl || 0;
        const newDaily = (strategy.dailyPnl || 0) + tradePnl;
        const newTotal = (strategy.totalPnl || 0) + tradePnl;

        // 4. Transactional update: Update strategy totals and mark trade as settled
        await db.transaction(async (tx) => {
          await tx.update(strategies)
            .set({ dailyPnl: newDaily, totalPnl: newTotal })
            .where(eq(strategies.id, strategy.id));

          await tx.update(strategyTrades)
            .set({ pnlCalculated: true })
            .where(eq(strategyTrades.id, trade.id));
        });

        log(`[SE] Settled Trade ID ${trade.id}. PnL: ${tradePnl}. Strategy ${strategy.id} updated.`);
      }
    } catch (error) {
      log(`[SE] Error processing unsettled trades: ${error}`);
    }
  }
}

export const settlementEngine = new SettlementEngine();
