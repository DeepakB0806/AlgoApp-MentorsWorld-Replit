2. Update server/routes/broker-routes.ts
Now we need to update the download route so it streams the file from the disk instead of trying to read it from the rawCsvCache that we just cleared.

Open server/routes/broker-routes.ts, add import fs from "fs"; and import path from "path"; at the top (if not already there), and locate the app.get("/api/broker-configs/:id/scrip-master-download") route (around line 1072).

Replace the entire route with this:

TypeScript
  app.get("/api/broker-configs/:id/scrip-master-download", async (req, res) => {
    try {
      const today = new Date();
      const dateStr = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, "0")}-${String(today.getDate()).padStart(2, "0")}`;
      const filename = `scrip_master_nfo_${dateStr}.csv`;
      const filePath = path.resolve(process.cwd(), filename);

      // Check if the file exists on disk
      if (!fs.existsSync(filePath)) {
        return res.status(503).json({ 
          error: "Scrip master file not found on disk. Please trigger a manual sync first to download today's file." 
        });
      }

      console.log(`[DOWNLOAD] Streaming file to client: ${filePath}`);

      // Set headers before piping
      res.setHeader('Content-Type', 'text/csv');
      res.setHeader('Content-Disposition', `attachment; filename="${filename}"`);
      
      // Stream the file directly from disk to the network response
      const fileStream = fs.createReadStream(filePath);
      
      fileStream.on('error', (error) => {
        console.error("[DOWNLOAD] Error streaming file:", error);
        if (!res.headersSent) {
          res.status(500).json({ error: "Failed to stream the file to the browser." });
        }
      });

      fileStream.pipe(res);

    } catch (error: any) {
      console.error("[DOWNLOAD] Route crash:", error);
      if (!res.headersSent) {
        res.status(500).json({ error: "Internal server error during download preparation." });
      }
    }
  });
Make these two changes and restart your server. The sync process will now write the file directly to your disk and instantly release the memory. When you click the Download button on the frontend, Node will stream it seamlessly without crashing.
