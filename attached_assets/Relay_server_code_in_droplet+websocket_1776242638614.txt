const express = require('express');
const axios = require('axios');
const httpProxy = require('http-proxy'); // Added for WebSockets

const app = express();

app.use(express.json({ limit: '1mb' }));
app.use(express.urlencoded({ extended: true, limit: '1mb' }));

// Create the WebSocket proxy server
const proxy = httpProxy.createProxyServer({
    ws: true,
    changeOrigin: true
});

// A secret key so only YOUR Replit app can use this relay
const RELAY_SECRET = "mentorsworld_algo_secure_2026";

// --- REST API PROXY (Your original logic) ---
app.use(async (req, res) => {
    // 1. Security Check
    if (req.headers['x-relay-secret'] !== RELAY_SECRET) {
        return res.status(403).send("Unauthorized Relay Access");
    }

    // 2. Extract the actual Kotak URL from the headers
    const targetUrl = req.headers['x-target-url'];
    if (!targetUrl) return res.status(400).send("Missing target URL");

    try {
        // 3. Remove relay-specific headers before forwarding to Kotak
        const forwardHeaders = { ...req.headers };
        delete forwardHeaders['host'];
        delete forwardHeaders['x-relay-secret'];
        delete forwardHeaders['x-target-url'];

        // 4. Forward the request to Kotak
        const response = await axios({
            method: req.method,
            url: targetUrl,
            headers: forwardHeaders,
            data: req.method === 'GET' ? undefined : req.body
        });

        // 5. Send Kotak's response back to Replit
        res.status(response.status).json(response.data);
    } catch (error) {
        if (error.response) {
            res.status(error.response.status).json(error.response.data);
        } else {
            res.status(500).json({ error: error.message });
        }
    }
});

const PORT = process.env.PORT || 8080;

// Capture the server instance so we can listen for WebSockets
const server = app.listen(PORT, '0.0.0.0', () => {
    console.log(`Relay server running on port ${PORT} (REST + WebSockets)`);
});

// --- WEBSOCKET PROXY (The Missing Piece) ---
server.on('upgrade', (req, socket, head) => {
    const secret = req.headers['x-relay-secret'];
    const targetUrl = req.headers['x-target-url'];

    // Security Check for WebSockets
    if (secret !== RELAY_SECRET || !targetUrl) {
        socket.write('HTTP/1.1 401 Unauthorized\r\n\r\n');
        socket.destroy();
        return;
    }

    // Forward the live WebSocket stream to Kotak
    proxy.ws(req, socket, head, { target: targetUrl });
});
