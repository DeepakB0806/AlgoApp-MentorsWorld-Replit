function buildHsmSubscriptionBinary(exchange: string, token: string): Buffer {
  // Kotak requires "nse_fo" for NFO tokens in HSM subscriptions
  const targetExchange = exchange.toLowerCase() === "nfo" ? "nse_fo" : exchange.toLowerCase();
  const scripStr = `${targetExchange}|${token}`;
  const scripLen = Buffer.byteLength(scripStr, 'utf8');

  // Total size: start(1) + type(1) + count(1) + [len(2)+scrip] + end(1)
  const totalLength = 1 + 1 + 1 + 2 + scripLen + 1;
  const buf = Buffer.alloc(totalLength);
  
  let offset = 0;
  buf.writeUInt8(10, offset++); // Start (0x0A)
  buf.writeUInt8(4, offset++);  // ReqTypeValues.SCRIP_SUBS (4)
  buf.writeUInt8(1, offset++);  // Field count (1)
  
  buf.writeUInt16BE(scripLen, offset); offset += 2;
  buf.write(scripStr, offset, scripLen, 'utf8'); offset += scripLen;
  
  buf.writeUInt8(13, offset++); // End (0x0D)
  return buf;
}

// 🔒 LOCKED BLOCK START — startHsmHeartbeat: 30 s ti heartbeat [HSM-4]
function startHsmHeartbeat(): void {
  if (heartbeatInterval) clearInterval(heartbeatInterval);
  heartbeatInterval = setInterval(() => {
    if (ws && ws.readyState === WebSocket.OPEN) {
      try {
        // TASK #245: Binary Heartbeat (THROTTLING_TYPE = 2)
        const buf = Buffer.alloc(4);
        buf.writeUInt8(10, 0); // Start
        buf.writeUInt8(2, 1);  // THROTTLING_TYPE
        buf.writeUInt8(0, 2);  // 0 fields
        buf.writeUInt8(13, 3); // End
        ws.send(buf);
      } catch {}
    }
  }, 30_000);
}
