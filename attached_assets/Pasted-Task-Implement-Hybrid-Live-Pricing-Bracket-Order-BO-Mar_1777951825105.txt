// ─── Hybrid Bracket Order HCM Block ─────────────────────────────────────────────
// Fetches live quotes via API. Falls back to CSV. Uses BO Margin Math.
// BO Sell Margin = (Strike + Premium) * Qty * 2.5%
// BO Buy Margin = Premium * Qty
async function cmHybridBOBlock(
  legs: any[],
  atmStrike: number,
  strikeInterval: number,
  tokenMap: Map<string, string>,
  priceMap: Map<number, { CE?: number; PE?: number }>,
  lotSize: number,
  lotMultiplier: number,
  brokerConfig: BrokerConfig,
  exchange: string,
  blockLabel: string,
): Promise<number> {
  const MLOG = "[MARGIN-CALC]";
  let totalMargin = 0;
  const BO_SELL_MARGIN_RATE = 0.025; // 2.5% rule for BO Selling

  for (const leg of legs) {
    const legType = (leg.type || "").toUpperCase() as "CE" | "PE" | "FUT";
    if (legType !== "CE" && legType !== "PE" && legType !== "FUT") continue;

    const legAction = (leg.action || "SELL").toUpperCase();
    const qty = (leg.lots || 1) * lotMultiplier * lotSize;

    // 1. Resolve Strike
    let resolvedStrike: number;
    if (legType === "FUT") {
      resolvedStrike = atmStrike;
    } else {
      const spec = parseStrikeSpec(leg.strike || "ATM");
      resolvedStrike = getOTMStrike(atmStrike, spec, strikeInterval, legType as "CE" | "PE");
    }

    // 2. Resolve Token
    let token = legType === "FUT" ? tokenMap.get(`0_FUT`) : tokenMap.get(`${resolvedStrike}_${legType}`);
    
    // 3. Resolve Premium (Try Live API first, fallback to CSV, fallback to 150)
    let premium = 0;
    let priceSource = "CSV";

    if (token && brokerConfig.isConnected && brokerConfig.accessToken) {
        try {
            // Note: Replace with EL.getQuotes array batching if supported, otherwise single getQuote
            const quoteRes = await EL.getQuote(brokerConfig, EL.mapExchange(exchange), token);
            if (quoteRes.success && quoteRes.ltp && quoteRes.ltp > 0) {
                premium = quoteRes.ltp;
                priceSource = "LIVE-API";
            }
        } catch (err) { /* ignore and fallback */ }
    }

    if (premium === 0) {
        const priceEntry = priceMap.get(resolvedStrike);
        premium = legType === "FUT" ? resolvedStrike : ((legType === "CE" ? priceEntry?.CE : priceEntry?.PE) ?? 150);
    }

    // 4. Calculate Bracket Order Math
    if (legAction === "SELL") {
      const sellReq = (resolvedStrike + premium) * qty * BO_SELL_MARGIN_RATE;
      totalMargin += sellReq;
      console.log(`${MLOG}   ${blockLabel} BO: SELL ${legType} @${resolvedStrike} prem=₹${premium.toFixed(2)} [${priceSource}] req=₹${sellReq.toFixed(2)}`);
    } else {
      const buyReq = premium * qty;
      totalMargin += buyReq;
      console.log(`${MLOG}   ${blockLabel} BO: BUY  ${legType} @${resolvedStrike} prem=₹${premium.toFixed(2)} [${priceSource}] req=₹${buyReq.toFixed(2)}`);
    }
    
    // Slight delay to prevent rate limits if making individual getQuote calls
    if (priceSource === "LIVE-API") await new Promise(r => setTimeout(r, 50));
  }
  
  return totalMargin; // Absolute total capital required for this BO block
}
